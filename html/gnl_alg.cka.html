  Fixpoint cKA_sat (l : list A) (e : reg A) : Prop :=
    match e with
    | 0r ⇒ False
    | 1r ⇒ l = []
    | r_var a ⇒ l = [a]
    | e ×r f ⇒ ∃ l1 l2, eq_list_comm l (l1 ++ l2) ∧ l1 |=cKA e ∧ l2 |=cKA f
    | e +r f ⇒ l |=cKA e ∨ l |=cKA f
    | e^* ⇒ ∃ L, eq_list_comm l (concat L) ∧ ∀ k, In k L → k |=cKA e
    end
  where " l |=cKA e " := (cKA_sat l e).
